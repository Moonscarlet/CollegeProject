This is the Arduino code we made to move the quadrotor:

#include

#define MAX_SIGNAL 1800
#define MIN_SIGNAL 800
Servo motors[4];
int x=27,PIN=8;
void setup()
{
Serial.begin(9600);
// Serial.println("Program begin...");
// Serial.println("This program will calibrate the ESC.");
//
for (int i=0; i<4; i++){
motors[i].attach(PIN);
PIN-=1;
} // Serial.println("Now writing maximum output.");
// Serial.println("Turn on power source, then wait 2 seconds and press any key.");

// // for (int i=0; i<4; i++){

} // // // Wait for input

// // // Send min output

// // for (int i=0; i<4; i++){
// motors[i].writeMicroseconds(MIN_SIGNAL);
} } void loop()
{
if (Serial.available() > 0){
int command = Serial.read();
if (command == '+'){
x+=1;
}
else if (command == '-'){
x-=1;
}
else if (command =='1'){
x=32;
}
else if (command =='0'){
x=27;
}
else if (command == '8')
{
Serial.println("Fly");
motors[0].write(x+10);
motors[1].write(x+10);
motors[2].write(x+10);
motors[3].write(x+10);
delay (500);
Serial.println("Balance");
}
else if (command == '2')
{
Serial.println("Land");
motors[0].write(x-10);
motors[1].write(x-10);
motors[2].write(x-10);
motors[3].write(x-10);
delay (500);
Serial.println("Balance");
}
else if (command == 'w')
{
Serial.println("Forward");
motors[0].write(x+10);
motors[1].write(x);
motors[2].write(x-10);
motors[3].write(x);
delay (500);
Serial.println("Balance");
}

else if (command == 's')
{
Serial.println("Back");
motors[0].write(x-10);
motors[1].write(x);
motors[2].write(x+10);
motors[3].write(x);
delay (500);
Serial.println("Balance");
}

else if (command == 'd')
{
Serial.println("Right");
motors[0].write(x);
motors[1].write(x+10);
motors[2].write(x);
motors[3].write(x-10);
delay (500);
Serial.println("Balance");
}

else if (command == 'a')
{
Serial.println("Left");
motors[0].write(x);
motors[1].write(x-10);
motors[2].write(x);
motors[3].write(x+10);
delay (500);
}
}

for (int i=0; i<4; i++){ motors[i].write(x);
}
Serial.println("");
Serial.println(motors[0].read());
delay(15);
}
